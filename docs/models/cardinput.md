# CardInput

An account with type `CARD` and subtype `DAILYPAY` or `DEBIT`.

## Example Usage

```typescript
import { CardInput } from "@dailypay/dailypay/models";

let value: CardInput = {
  name: "Debit Card",
  accountType: "CARD",
  subtype: "DEBIT",
  cardAccountDetails: {
    token: "HX46TY794RG",
    issuer: "411600",
    firstName: "Edith",
    lastName: "Clarke",
    expirationMonth: "02",
    expirationYear: "2025",
    addressLineOne: "123 Kebly Street",
    addressLineTwo: "Apt #12",
    addressCity: "Fort Lee",
    addressState: "NJ",
    addressZipCode: "72374",
    addressCountry: "US",
  },
};
```
